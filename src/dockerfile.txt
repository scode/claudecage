FROM ubuntu:24.04

ARG USERNAME
ARG UID
ARG GID
ARG HOST_HOME

RUN apt-get update \
    && apt-get install -y --no-install-recommends \
        bubblewrap \
        build-essential \
        ca-certificates \
        curl \
        file \
        git \
        imagemagick \
        less \
        procps \
        sudo \
        unzip \
        zip \
    && curl -fsSL https://deb.nodesource.com/setup_22.x | bash - \
    && apt-get install -y --no-install-recommends nodejs \
    && apt-get clean \
    && rm -rf /var/lib/apt/lists/*

RUN groupadd -g ${GID} ${USERNAME} 2>/dev/null; \
    userdel -f $(id -nu ${UID} 2>/dev/null) 2>/dev/null; \
    useradd -u ${UID} -g ${GID} -m -s /bin/bash -o ${USERNAME} \
    && echo "${USERNAME} ALL=(ALL) NOPASSWD:ALL" > /etc/sudoers.d/brew-install

# Install Homebrew as non-root user (needs sudo for /home/linuxbrew ownership)
USER ${USERNAME}
RUN NONINTERACTIVE=1 /bin/bash -c "$(curl -fsSL https://raw.githubusercontent.com/Homebrew/install/HEAD/install.sh)"
ENV PATH="/home/${USERNAME}/.local/bin:/home/linuxbrew/.linuxbrew/bin:/home/linuxbrew/.linuxbrew/sbin:${PATH}" \
    LD_LIBRARY_PATH="/home/linuxbrew/.linuxbrew/lib"

RUN brew install dprint gh sapling scode/dist-tap/leiter rust uv
RUN uv tool install ghstack

# Remove temporary sudo grant, configure system-level git settings, and
# create a symlink from the host home path to the container home path.
# Some tools (e.g., Claude Code plugins) persist absolute host paths in
# config files; the symlink lets those paths resolve inside the container.
USER root
RUN rm /etc/sudoers.d/brew-install
RUN CONTAINER_HOME="/home/${USERNAME}" \
    && if [ "${HOST_HOME}" != "${CONTAINER_HOME}" ]; then \
         mkdir -p "$(dirname "${HOST_HOME}")" \
         && ln -s "${CONTAINER_HOME}" "${HOST_HOME}"; \
       fi

# Configure gh as git credential helper so git push works via GH_TOKEN
# without user setup. Written directly because `gh auth setup-git` requires
# an authenticated session.
#
# This must be --system (not --global) because the host's ~/.gitconfig is
# bind-mounted read-only into the container at runtime (so git picks up the
# user's identity and aliases). That mount shadows the image's ~/.gitconfig,
# which is where --global writes. System config (/etc/gitconfig) is not
# shadowed and takes effect alongside the mounted user config.
RUN git config --system credential.https://github.com.helper '!/home/linuxbrew/.linuxbrew/bin/gh auth git-credential'

# Declared here so `image refresh` invalidates only the refreshable tail.
ARG REFRESH_MARKER=stable
RUN echo "${REFRESH_MARKER}" > /dev/null \
    && npm install -g @anthropic-ai/claude-code@latest @openai/codex@latest

USER ${USERNAME}
RUN echo "${REFRESH_MARKER}" > /dev/null && brew update && brew install cesarferreira/tap/stax
