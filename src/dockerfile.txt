FROM ubuntu:24.04

ARG USERNAME
ARG UID
ARG GID

RUN apt-get update \
    && apt-get install -y --no-install-recommends \
        build-essential \
        ca-certificates \
        curl \
        file \
        git \
        procps \
        sudo \
    && curl -fsSL https://deb.nodesource.com/setup_22.x | bash - \
    && apt-get install -y --no-install-recommends nodejs \
    && npm install -g @anthropic-ai/claude-code@latest \
    && apt-get clean \
    && rm -rf /var/lib/apt/lists/*

RUN groupadd -g ${GID} ${USERNAME} 2>/dev/null; \
    userdel -f $(id -nu ${UID} 2>/dev/null) 2>/dev/null; \
    useradd -u ${UID} -g ${GID} -m -s /bin/bash -o ${USERNAME} \
    && echo "${USERNAME} ALL=(ALL) NOPASSWD:ALL" > /etc/sudoers.d/brew-install

# Install Homebrew as non-root user (needs sudo for /home/linuxbrew ownership)
USER ${USERNAME}
RUN NONINTERACTIVE=1 /bin/bash -c "$(curl -fsSL https://raw.githubusercontent.com/Homebrew/install/HEAD/install.sh)"
ENV PATH="/home/linuxbrew/.linuxbrew/bin:/home/linuxbrew/.linuxbrew/sbin:${PATH}"

RUN brew install gh scode/dist-tap/leiter

# Configure gh as git credential helper so git push uses GH_TOKEN when set.
RUN gh auth setup-git

# Remove temporary sudo grant
USER root
RUN rm /etc/sudoers.d/brew-install

USER ${USERNAME}
